Some simple list expressions
  • [ 1 2 3 ] = 1, 2, 3 = [1, 2, 3]
  • [ 1 2 3 number1 ] = 1, 2, 3, 1 = [1, 2, 3, 1]
  • [ 'jaja' 'jeje' 3 number1] = jaja, jeje, 3, 1 = [jaja, jeje, 3, 1]
List expressions with ranges
  • [ 1:3 ] = 1, 2, 3 = [1, 2, 3]
  • [ number1:number3 'jaja' number1 ] = 1, 2, 3, 'jaja', 1 = [1, 2, 3, jaja, 1]
  • [ :3 ] = 0, 1, 2, 3 = [0, 1, 2, 3]
  • [ :number3 ] = 0, 1, 2, 3 = [0, 1, 2, 3]
  • [ 1:5:2 ] = 1, 3, 5 = [1, 3, 5]
  • [ 1:6:2 ] = 1, 3, 5 = [1, 3, 5]
  • [ :5:2 ] = 0, 2, 4 = [0, 2, 4]
  • [ :number5:number2 ] = 0, 2, 4 = [0, 2, 4]
  • [ -3:5:2 ] = -3, -1, 1, 3, 5 = [-3, -1, 1, 3, 5]
  • [ 5:-3:-2 ] = 5, 3, 1, -1, -3 = [5, 3, 1, -1, -3]
  • [ 5:-4:-2 ] = 5, 3, 1, -1, -3 = [5, 3, 1, -1, -3]
  • [ :-5:-2 ] = 0, -2, -4 = [0, -2, -4]
  • [ :-5:number-2 ] = 0, -2, -4 = [0, -2, -4]
Arithmethic expressions using lists
  • +: [ 1 2 3 ] = 1 + 2 + 3 = 6 = 6
  • +: [ 1:3 ] = 1 + 2 + 3 = 6 = 6
  • +: [ 1:3 10 ] = 1 + 2 + 3 + 10 = 16 = 16
  • +: from1To3 = 1 + 2 + 3 = 6 = 6
  • +: from1To3 10 = 1 + 2 + 3 + 10 = 16 = 16
  • +: [ number1 1:3 ] = 1 + 1 + 2 + 3 = 7 = 7
  • +: from1To3 [ 1:3 ] = (1 + 2 + 3) * 2 = 12 = 12
  • +: from1To3 [ 1:3 ] 10 = (1 + 2 + 3) * 2 + 10 = 22 = 22
